## 다른 사람의 풀이

1. replace함수 활용
```js
function solution(my_string) {
    return my_string.replace(/[aeiou]/g, '');
}
```

2. 배열을 만들지 않고 바로 사용
```js
function solution(my_string) {
    return Array.from(my_string).filter(t => !['a', 'e', 'i', 'o', 'u'].includes(t)).join('');
}
```

3. 반복문을 사용
```js
function solution(my_string) {
    let answer = '';
    let alpa=['a','e','i','o','u']
    for(let i=0;i<my_string.length;i++){
        if(!alpa.includes(my_string[i])){
            answer+=my_string[i]
        }
    }
    return answer;
}
```
